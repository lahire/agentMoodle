<script>
  // Generate a random session ID for this browser tab so the bot remembers the context
  const sessionId = "sess_" + Math.random().toString(36).substring(2, 10);
  // Ideally, grab the real Moodle user ID, but we'll use a random hash for testing
  const userHash = "moodle_user_" + Math.floor(Math.random() * 10000); 

  const btnOpen = document.getElementById('ai-chat-btn');
  const btnClose = document.getElementById('ai-chat-close');
  const chatWindow = document.getElementById('ai-chat-window');
  const btnSend = document.getElementById('ai-chat-send');
  const inputField = document.getElementById('ai-chat-input');
  const messagesBox = document.getElementById('ai-chat-messages');
  const confidenceSelect = document.getElementById('ai-confidence');

  // Toggle Window
  btnOpen.addEventListener('click', () => {
    chatWindow.style.display = chatWindow.style.display === 'flex' ? 'none' : 'flex';
  });
  btnClose.addEventListener('click', () => chatWindow.style.display = 'none');

  // Append Message Helper
  function addMessage(text, sender) {
    const msgDiv = document.createElement('div');
    msgDiv.classList.add('chat-msg', sender === 'user' ? 'msg-user' : 'msg-bot');
    msgDiv.innerText = text;
    messagesBox.appendChild(msgDiv);
    messagesBox.scrollTop = messagesBox.scrollHeight;
  }

  // Send Message Logic
  async function sendMessage() {
    const question = inputField.value.trim();
    if (!question) return;

    const confidence = confidenceSelect.value;
    inputField.value = '';
    
    // Add user message to UI
    addMessage(question, 'user');

    // Add Loading indicator
    const loadingDiv = document.createElement('div');
    loadingDiv.classList.add('chat-msg', 'msg-loading');
    loadingDiv.innerText = "Pensando...";
    messagesBox.appendChild(loadingDiv);
    messagesBox.scrollTop = messagesBox.scrollHeight;

    try {
      const response = await fetch("http://localhost:8000/ask", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          user_hash: userHash,
          session_id: sessionId,
          question: question,
          confidence_level: confidence
        })
      });

      const data = await response.json();
      messagesBox.removeChild(loadingDiv); // Remove loading
      addMessage(data.answer, 'bot');      // Add AI answer

    } catch (error) {
      messagesBox.removeChild(loadingDiv);
      addMessage("Error de conexión con el servidor. Por favor, intenta de nuevo.", 'bot');
      console.error(error);
    }
  }

  // Event Listeners for sending
  btnSend.addEventListener('click', sendMessage);
  inputField.addEventListener('keypress', (e) => {
    if (e.key === 'Enter') sendMessage();
  });
</script>
